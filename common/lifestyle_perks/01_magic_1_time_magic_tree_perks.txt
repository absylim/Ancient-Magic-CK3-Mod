time_mage_perk = {
	lifestyle = arcane_lifestyle
	tree = ancient_magic_time
	position = { 2 0 }
	icon = node_diplomacy

	parent = mage_perk

	auto_selection_weight = {
		value = 11
		if = {
			limit = {
				OR = {
					has_trait = magic_potential
					has_trait = education_arcane
				}
			}
			add = 1989
		}
		if = {
			limit = {
				has_trait = ambitious
			}
			multiply = 2
		}
		if = {
			limit = {
				has_trait = greedy
			}
			multiply = 2
		}
		if = {
			limit = {
				has_trait = temperate
			}
			multiply = 2
		}
		if = {
			limit = {
				has_trait = patient
			}
			multiply = 2
		}
		if = {
			limit = {
				NOT = { is_immortal=yes }
			}
			multiply = 2
		}
		if = {
			limit = {
				can_start_new_lifestyle_tree_trigger = no
			}
			multiply = 0
		}
	}

	character_modifier = {
		monthly_income_mult = 0.1
	}

	effect = {
		gain_mana_gen_perk = { VALUE = time_mage_perk_gen_cost }
		gain_max_mana_perk = { VALUE = time_mage_perk_max_mana }
		mod_bit = { MASK = time_perk_mask_0 BIT = time_mage_perk_bit OP = set }
		trigger_event = { on_action = on_reset_mana_system }
	}
}
