life_mage_perk = {
	lifestyle = ancient_magic_lifestyle
	tree = life_magic
	position = { 2 0 }
	icon = node_diplomacy

	auto_selection_weight = {
		value = 11
		if = {
			limit = {
				OR = {
					has_trait = magic_potential
					has_trait = education_arcane
				}
			}
			add = 1989
		}
		if = {
			limit = {
				has_trait = kind
			}
			multiply = 2
		}
		if = {
			limit = {
				has_trait = gregarious
			}
			multiply = 2
		}
		if = {
			limit = {
				has_trait = compassionate
			}
			multiply = 2
		}
		if = {
			limit = {
				can_start_new_lifestyle_tree_trigger = no
			}
			multiply = 0
		}
	}

	character_modifier = {
		health = 0.1
	}

	effect = {
		gain_mana_gen_perk = { VALUE = life_mage_perk_gen_cost }
		gain_max_mana_perk = { VALUE = life_mage_perk_max_mana }
		make_mage_effect = yes
		mod_bit = { BIT = life_mage_perk_bit OP = set }
		trigger_event = { on_action = on_reset_mana_system }
	}
}
