mana_affinity_level = {
	type = character

	text = {
		localization_key = mana_affinity_level_mundane
		trigger = { mana_affinity <= mundane_mana_affinity_level }
	}

	text = {
		localization_key = mana_affinity_level_sub_magical
		trigger = {
			mana_affinity > mundane_mana_affinity_level
			mana_affinity <= sub_magical_mana_affinity_level
		}
	}

	text = {
		localization_key = mana_affinity_level_magical
		trigger = {
			mana_affinity > sub_magical_mana_affinity_level
			mana_affinity <= magical_mana_affinity_level
		}
	}

	text = {
		localization_key = mana_affinity_level_loved
		trigger = {
			mana_affinity > magical_mana_affinity_level
			mana_affinity <= loved_mana_affinity_level
		}
	}

	text = {
		localization_key = mana_affinity_level_demigod
		trigger = {
			mana_affinity > magical_mana_affinity_level
			mana_affinity <= demigod_mana_affinity_level
		}
	}

	text = {
		localization_key = mana_affinity_level_god
		trigger = { mana_affinity > demigod_mana_affinity_level }
	}
}

		trigger = { mana_blood_compatibility_with_owner < terrible_mana_blood_compatibility_level }

	text = {
		localization_key = skill_level_poor

			mana_blood_compatibility_with_owner >= terrible_mana_blood_compatibility_level

		}
	}

	text = {
		localization_key = skill_level_average

			mana_blood_compatibility_with_owner >= average_mana_blood_compatibility_level
			mana_blood_compatibility_with_owner < good_mana_blood_compatibility_level
		}
	}

	text = {
		localization_key = skill_level_good
		trigger = {
			mana_blood_compatibility_with_owner >= good_mana_blood_compatibility_level
			mana_blood_compatibility_with_owner < excellent_mana_blood_compatibility_level
		}
	}

	text = {
		localization_key = skill_level_excellent
		trigger = { mana_blood_compatibility_with_owner >= excellent_mana_blood_compatibility_level }
	}
}
