mana_blood_in_inventory = {
	base = character_artifact
	conditions = {
		has_artifact_feature_group = mana_blood
	}
}

mana_blood_in_inventory_close_family = { #Blood of close family members - High compatibility
	base = mana_blood_in_inventory
	conditions = {
		scope:creator.is_close_family_of = owner
	}
}

mana_blood_in_inventory_extended_family = { #Blood of close family members - Medium compatibility
	base = mana_blood_in_inventory
	conditions = {
		scope:creator.is_extended_family_of = owner
	}
}

mana_blood_in_inventory_others = {  #Blood of non-family members or very distant ones - Low compatibility
	base = mana_blood_in_inventory
	conditions = {
		NOT = {
			scope:creator.is_close_family_of = owner
			scope:creator.is_extended_family_of = owner
		}
	}
}
