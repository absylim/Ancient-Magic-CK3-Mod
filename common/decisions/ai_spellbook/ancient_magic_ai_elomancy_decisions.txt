stone_golem_ai_spell_decision = {
	is_shown = {
		is_ai = yes
		is_magus = yes
		has_perk = golem_creation_perk
		var:mana.compare_value >= var:golem_creation_spell_cost.compare_value
		NOT = { has_character_modifier = golem_cooldown_modifier }
	}
	
	ai_check_frequency = 1	
	
	effect = {
		deduct_mana_with_tooltip = { VALUE = var:golem_creation_spell_cost }
		spawn_army = {
			name = golem_army
			uses_supply = no
			inheritable = no
			men_at_arms = {
				type = golem
				stacks = mage_level
			}
			location = capital_province
		}
		add_character_modifier = {
			modifier = golem_cooldown_modifier
			years = 5
		}
	}
	
	ai_potential = { always = yes }
	
	ai_will_do = {
		base = 0
		
		modifier = {
			add = 30
			is_at_war = yes
		}
	}
}

flame_aura_ai_spell_decision = {
	is_shown = {
		is_ai = yes
		is_magus = yes
		has_perk = flame_aura_perk
		var:mana.compare_value >= var:flame_aura_spell_cost.compare_value
	}
	
	ai_check_frequency = 1	
	
	effect = {
		if = {
			limit = {
				var:mana.compare_value >= var:flame_aura_spell_cost.compare_value
				NOT = {
					has_character_modifier = flame_aura_modifier
				}
			}

			deduct_mana_with_tooltip = { VALUE = var:flame_aura_spell_cost }

			change_mana_gen = { VALUE = -15 }

			while = {
				count = mage_level
				add_character_modifier = {
					modifier = flame_aura_modifier
					mode = add
				}
			}
		}
		else_if = {
			limit = {
				OR = {
					has_character_modifier = flame_aura_modifier
				}
			}

			change_mana_gen = { VALUE = 15 }

			while = {
				count = 100
				remove_character_modifier = flame_aura_modifier
			}
		}
	}
	
	ai_potential = { always = yes }
	
	ai_will_do = {
		base = 25
			
		modifier = {
			factor = 0
			var:mana_gen < 15
		}
		
		modifier = {
			factor = 0
			has_character_modifier = flame_aura_modifier
		}
		
		modifier = {
			add = 50
			has_character_modifier = flame_aura_modifier
			var:mana_gen <= 0
		}
	}
}
