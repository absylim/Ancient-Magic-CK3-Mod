# Valid OPs: flip, set, unset
mod_bit = {
	if = {
		limit = { NOT = { exists = var:$MASK$ } }
		set_variable = {
			name = $MASK$
			value = $MASK$_maker
		}
	}
	set_variable = {
		name = $MASK$
		value = var:$MASK$.$BIT$_$OP$
	}
}
