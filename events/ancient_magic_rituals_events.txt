namespace = ancient_magic_rituals_events

ancient_magic_rituals_events.001 = { #Forge soul
	title = ancient_magic_rituals_events.001.t
	desc = ancient_magic_rituals_events.001.desc
		
	option = {
		trigger = {
			var:mana >= var:soul_materialisation_mana_cost
			exists = mage_level
		}
		name = ancient_magic_rituals_events.001.a
		
		change_mana = { VALUE = var:soul_materialisation_mana_cost }
		
		random_list = {
			50 = {
				modifier = {add = {value = mage_level subtract = 15}} # mage level 15 is minimum for tipping to succcess
			}
			50 = {
				modifier = {add = {value = 15 subtract = mage_level}}
			}
		}
		
	}
	
	option = { #Give up
		trigger = {
			is_ai = no
		}
		name = ancient_magic_rituals_events.001.quit
	}
}
